//*********** Solution Thinking Idea ************
// 1. Plan to get the reverse of the number using arithmetic operations. 
// 2. Compare the generated number with the input number.   


class Solution {
    public boolean isPalindrome(int x) {
        int num = x;
        int rev = 0;
        if(x < 0 ) return false;
        while(num!=0){
            rev = rev*10 +num%10;
            num /= 10;
        }
        return (rev==x);
    }
}


// Time Complexity    = O(d)  where d is the digits in x; O(log x)   
// Space Complexity   = O(1)
