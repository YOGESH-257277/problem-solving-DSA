***************************************************************************************
// Method 04    //(MANUAL)    //CHARACTER STRING TO ARRAY
String s = "hello";
char[] arr = new char[s.length()];

for (int i = 0; i < s.length(); i++) {
    arr[i] = s.charAt(i);
}

***************************************************************************************
//METHOD 05                  //NUMBER STRING TO ARRAY
String s = "1234";
int[] arr = new int[s.length()];

for (int i = 0; i < s.length(); i++) {
    arr[i] = s.charAt(i) - '0';
}
