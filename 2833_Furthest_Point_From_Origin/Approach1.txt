//*********** Solution Thinking Idea ************
// 1. Using Mathematical
// 2. Using String to Array Conversion

class Solution {
    public int furthestDistanceFromOrigin(String moves) {
        int cnt = 0 ;
        int lcnt = 0 ; 
        int rcnt = 0;
        int spaces = 0;

        for(char ch : moves.toCharArray()){
            if(ch=='L') lcnt++;
            else if(ch=='R') rcnt++;
            else spaces++;
        }
        if(lcnt >= rcnt){
            return (lcnt - rcnt)+spaces;
        }
        else return (rcnt - lcnt)+ spaces;
    }
}


// Time Complexity    = O(n),  where n is the total length of string moves.   
// Space Complexity   = O(1) , Only used for variables.
