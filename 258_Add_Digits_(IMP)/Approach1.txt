// Reverse and Remember the Mathematical Expression

class Solution {
    public int addDigits(int num) {
        if(num == 0) return 0;
        if(num%9==0) return 9;
        else return num%9;
    }
}

// Time Complexity = O(1), For number
// Space Complexity = O(1) No extra space needed.
