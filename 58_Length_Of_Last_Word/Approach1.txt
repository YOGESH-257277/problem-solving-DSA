//*********** Solution Thinking Idea ************
// 1. Check element != empty space the increment count.
// 2. Check cnt > 0 means we have already found the elements. break it.
// 3. If cnt is zero continue until we found some non empty thing like char.

class Solution {
    public int lengthOfLastWord(String s) {
        int n = s.length();
        int cnt = 0;

        for(int i = n-1 ; i >= 0 ; i--){
            char ch = s.charAt(i);
            if(ch !=' ') cnt++;
            else if(cnt > 0) break;
            else continue;
        }
        return cnt;
    }
}

// Time Complexity    = O(n), where n is the length of the given string s.   
// Space Complexity   = O(1), variables.
