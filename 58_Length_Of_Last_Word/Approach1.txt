class Solution {
    public int lengthOfLastWord(String s) {
        int n = s.length();
        int cnt = 0;

        for(int i = n-1 ; i >= 0 ; i--){
            char ch = s.charAt(i);
            if(ch !=' ') cnt++;
            else if(cnt > 0) break;
            else continue;
        }
        return cnt;
    }
}

// Time Complexity    = O(n), where n is the length of the given string s.   
// Space Complexity   = O(1), variables.
