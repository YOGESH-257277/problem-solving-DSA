//*********** Solution Thinking Idea ************
// 1.  Check the base case.
// 2.  Initialize a variable k = 1.
// 3.  Iterate through the array starting from the second element.
// 4.  If the current element is not equal to the previous element, assign the current element to the kth position in the array and increment k.
// 5.  Return the value of k as the new length of the array without duplicates.

class Solution {
    public int removeDuplicates(int[] nums) {
        if(nums.length == 0) return 0;

        int k = 1;
        for(int i = 1 ; i < nums.length ; i++){
            if(nums[i]!= nums[i-1]){
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}

// Time Complexity    = O(n), where n is the length of the given array   
// Space Complexity   = O(1), Pointer
