//*********** Solution Thinking Idea ************
// 1. Mathematical Approach Sum of n number = n*(n+1)/2, where n is the length of nums array.
// 2. Find the sum of all the arrays in the nums array. 
// 3. Subtract the Total_Required_Sum which is n*(n+1)/2 from the sum of array elements to get the missing number. 


class Solution {
    public int missingNumber(int[] nums) {
        int total = nums.length;
        int desiredsum = (total*(total+1))/2;
        int actualsum =0;
        for(int i : nums){
            actualsum += i;
        }
        return desiredsum-actualsum;
    }
}


// Time Complexity    = O(n), where n is the length of the nums array   
// Space Complexity   = O(1)
