//*********** Solution Thinking Idea ************
// 1. Check the Base case in case of tree.
// 2. Check the root value with left and right value. if not equal return false.
// 3. Check the left and right subtree recursively.

// NOTE: Think of recursively recursion to reach all the nodes. 

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isUnivalTree(TreeNode root) {
        
        if(root == null) return true;

        if(root.left != null && root.left.val != root.val ){ return false;}
        if(root.right != null && root.right.val != root.val){ return false;}

        return isUnivalTree(root.left) && isUnivalTree(root.right);

    }
}

// Time Complexity    = O(n), where n is the length tree all the nodes traversal 
// Space Complexity   = O(h), levels of trees/ height of tree.
